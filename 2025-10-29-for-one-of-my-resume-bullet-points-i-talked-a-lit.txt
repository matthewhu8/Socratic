  @app.post("/api/smart-practice/generate-session")
  async def generate_smart_practice_session(
      request: Dict,
      current_user: StudentUser = Depends(get_current_student),
      db: Session = Depends(get_db)
  ):
      """
      Generate a personalized practice session using MCP
      
      Flow:
      1. Analyze student's knowledge profile
      2. Identify weak skills via MCP analytics
      3. Fetch targeted questions via MCP question bank
      4. Sequence questions by difficulty via MCP recommendations
      5. Return adaptive problem set
      """
      try:
          # Get current knowledge profile
          profile = KnowledgeProfileService.get_student_profile(db, current_user.id)

          # Initialize MCP clients
          analytics_mcp = LearningAnalyticsMCP()
          question_mcp = QuestionBankMCP()
          recommendation_mcp = RecommendationMCP()

          # Step 1: Identify target skills (weakest skills)
          target_skills = await analytics_mcp.identify_weak_skills(
              profile=profile,
              subject=request.get("subject"),
              threshold=60  # Skills below 60% proficiency
          )

          # Step 2: For each weak skill, get prerequisite chain
          learning_path = []
          for skill in target_skills:
              prerequisites = await recommendation_mcp.get_prerequisite_chain(skill)
              learning_path.extend(prerequisites)

          # Step 3: Generate question sequence
          practice_questions = []
          for skill_node in learning_path:
              # Determine target difficulty based on current proficiency
              current_score = get_skill_score(profile, skill_node["name"])
              target_difficulty = calculate_zone_of_proximal_development(current_score)

              # Fetch 3-5 questions via MCP
              questions = await question_mcp.get_questions_by_skill(
                  skill_name=skill_node["name"],
                  difficulty=target_difficulty,
                  count=3,
                  sources=["ncert", "khan_academy", "brilliant"]  # MCP connects to all
              )

              practice_questions.extend(questions)

          # Step 4: Create session with adaptive sequencing
          session = SmartPracticeSession(
              user_id=current_user.id,
              questions=practice_questions,
              learning_objectives=target_skills,
              adaptive_rules={
                  "increase_difficulty_after_n_correct": 2,
                  "provide_hint_after_n_incorrect": 2,
                  "switch_skill_after_mastery": True
              }
          )

          db.add(session)
          db.commit()

          return {
              "session_id": session.id,
              "questions": practice_questions[:5],  # Return first 5
              "learning_path": learning_path,
              "estimated_time_minutes": len(practice_questions) * 5,
              "objectives": target_skills
          }

      except Exception as e:
          raise HTTPException(status_code=500, detail=str(e))

  Real-time Adaptation During Practice
  @app.post("/api/smart-practice/submit-answer")
  async def submit_smart_practice_answer(
      request: Dict,
      current_user: StudentUser = Depends(get_current_student),
      db: Session = Depends(get_db)
  ):
      """
      Process answer and adapt next question in real-time
      
      This is where MCP shines - pulling questions on-demand
      based on student performance
      """
      session_id = request.get("session_id")
      question_id = request.get("question_id")
      answer_data = request.get("answer")

      # Grade the answer
      grading_result = await grade_answer(question_id, answer_data)

      # Update knowledge profile
      updated_profile = KnowledgeProfileService.update_profile_after_grading(
          db, current_user.id, question_id, "smart-practice", grading_result
      )

      # HERE'S THE MCP MAGIC: Adaptive next question selection
      analytics_mcp = LearningAnalyticsMCP()
      question_mcp = QuestionBankMCP()

      # Analyze performance trend
      performance_analysis = await analytics_mcp.analyze_recent_performance(
          user_id=current_user.id,
          session_id=session_id,
          latest_result=grading_result
      )

      # Get next optimal question via MCP
      if performance_analysis["streak"] >= 3:
          # Student doing well, increase difficulty
          next_question = await question_mcp.get_next_adaptive_question(
              current_skill=performance_analysis["current_skill"],
              difficulty_adjustment=+0.2,
              student_profile=updated_profile
          )
      elif performance_analysis["struggles"] >= 2:
          # Student struggling, provide scaffolding
          next_question = await question_mcp.get_scaffolding_question(
              problematic_skill=performance_analysis["weak_skill"],
              difficulty=0.3,  # Easier problem
              include_hints=True
          )
      else:
          # Continue current progression
          next_question = await question_mcp.get_next_in_sequence(
              session_id=session_id
          )

      return {
          "grading_result": grading_result,
          "next_question": next_question,
          "performance_insights": performance_analysis,
          "progress": {
              "questions_completed": session.questions_answered,
              "mastery_gained": calculate_mastery_delta(updated_profile)
          }
      }

      async def _get_student_profile(self, user_id: int) -> Dict:

      async def _search_questions(self, skill: str, difficulty_min: float, 
                                  difficulty_max: float, topic: str = None, 
                                  limit: int = 10) -> Dict:
          """Search questions in database"""
          # Query across NCERT Examples, Exercises, and PYQs
          from sqlalchemy import or_, and_

          query = self.db.query(NcertExercises).filter(
              and_(
                  NcertExercises.difficulty >= difficulty_min,
                  NcertExercises.difficulty <= difficulty_max
              )
          )

          if topic:
              query = query.filter(NcertExercises.topic == topic)

          # Filter by skill in skills_tested JSON
          results = query.limit(limit).all()

          questions = []
          for q in results:
              if q.skills_tested and 'skills' in q.skills_tested:
                  for skill_data in q.skills_tested['skills']:
                      if skill.lower() in skill_data.get('skill_name', '').lower():
                          questions.append({
                              "id": q.id,
                              "question_text": q.question_text,
                              "difficulty": q.difficulty,
                              "topic": q.topic,
                              "skills_tested": q.skills_tested
                          })
                          break

          return {"questions": questions, "count": len(questions)}

      async def _analyze_recent_performance(self, user_id: int, n_recent: int) -> Dict:
          """Analyze recent grading sessions"""
          recent_sessions = self.db.query(GradingSession).filter(
              GradingSession.user_id == user_id,
              GradingSession.status == "completed"
          ).order_by(GradingSession.created_at.desc()).limit(n_recent).all()

          if not recent_sessions:
              return {"error": "No recent sessions found"}

          correct_count = 0
          total_score = 0
          difficulties = []

          for session in recent_sessions:
              if session.grading_result and 'grade' in session.grading_result:
                  grade_parts = session.grading_result['grade'].split('/')
                  score = float(grade_parts[0]) / 10.0
                  total_score += score
                  if score >= 0.7:
                      correct_count += 1

          return {
              "n_analyzed": len(recent_sessions),
              "accuracy": correct_count / len(recent_sessions),
              "average_score": total_score / len(recent_sessions),
              "streak": self._calculate_streak(recent_sessions)
          }

      def _calculate_streak(self, sessions: List) -> int:
          """Calculate current correct streak"""
          streak = 0
          for session in sessions:
              if session.grading_result and 'grade' in session.grading_result:
                  grade_parts = session.grading_result['grade'].split('/')
                  score = float(grade_parts[0]) / 10.0
                  if score >= 0.7:
                      streak += 1
                  else:
                      break
          return streak

      async def _calculate_zpd_difficulty(self, current_skill_score: float, 
                                         recent_accuracy: float) -> Dict:
          """Calculate optimal difficulty"""
          # Zone of Proximal Development algorithm
          base_difficulty = current_skill_score / 100.0

          if recent_accuracy >= 0.8:
              # Doing well, increase difficulty
              target = min(1.0, base_difficulty + 0.15)
          elif recent_accuracy >= 0.6:
              # Steady performance, slight increase
              target = min(1.0, base_difficulty + 0.05)
          elif recent_accuracy >= 0.4:
              # Struggling, maintain level
              target = base_difficulty
          else:
              # Failing, decrease difficulty
              target = max(0.1, base_difficulty - 0.15)

          return {
              "target_difficulty": target,
              "reasoning": f"Based on skill score {current_skill_score}% and accuracy {recent_accuracy}%"
          }

      async def _identify_skill_gaps(self, user_id: int, target_skill: str) -> Dict:
          """Identify weak prerequisite skills"""

      async def _predict_success_probability(self, user_id: int, question_id: int) -> Dict:
          """Predict success using simple heuristic (could be ML model)"""
          user = self.db.query(StudentUser).filter(StudentUser.id == user_id).first()
          question = self.db.query(NcertExercises).filter(NcertExercises.id == question_id).first()

          if not question or not user.knowledge_profile:
              return {"probability": 0.5, "confidence": "low"}

          # Simple heuristic: compare skill score to question difficulty
          question_difficulty = question.difficulty or 0.5

          # Get relevant skill scores
          relevant_scores = []
          if question.skills_tested and 'skills' in question.skills_tested:
              for skill_data in question.skills_tested['skills']:
                  skill_name = skill_data.get('skill_name')
                  # Find this skill in profile
                  for topic in user.knowledge_profile.get('subjects', {}).get('mathematics', {}).get('topics', []):
                      if skill_name in topic.get('skills', {}):
                          relevant_scores.append(topic['skills'][skill_name]['score'] / 100.0)

          if relevant_scores:
              avg_skill = sum(relevant_scores) / len(relevant_scores)
              # Simple probability model
              probability = max(0.1, min(0.9, avg_skill - question_difficulty + 0.5))
              return {"probability": probability, "confidence": "medium"}

          return {"probability": 0.5, "confidence": "low"}

  2. MCP Client (Orchestrates the back-and-forth with Claude)

  # Backend/app/services/mcp_client.py

  from anthropic import Anthropic
  from typing import List, Dict
  import json

  class MCPClient:
      """
      Client that sends requests to Claude with MCP tools.
      Claude decides which tools to call.
      """

      def __init__(self, mcp_server: EducationMCPServer):
          self.anthropic = Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))
          self.mcp_server = mcp_server

      async def query(self, user_prompt: str, max_iterations: int = 10) -> Dict:
          """
          Send a query to Claude with access to MCP tools.
          Claude will call tools as needed.
          
          This is THE KEY METHOD - this is where Claude decides!
          """

          messages = [
              {
                  "role": "user",
                  "content": user_prompt
              }
          ]

          iteration = 0

          while iteration < max_iterations:
              # Call Claude API with tool definitions
              response = self.anthropic.messages.create(
                  model="claude-3-5-sonnet-20241022",
                  max_tokens=4096,
                  tools=self.mcp_server.get_tools_schema(),  # ‚Üź Tools available to Claude
                  messages=messages
              )

              # Check if Claude wants to use tools
              if response.stop_reason == "tool_use":
                  # Claude decided to call one or more tools!
                  tool_results = []

                  for content_block in response.content:
                      if content_block.type == "tool_use":
                          tool_name = content_block.name
                          tool_input = content_block.input

                          print(f"ūü§Ė Claude decided to call: {tool_name}")
                          print(f"   with input: {json.dumps(tool_input, indent=2)}")

                          # Execute the tool
                          tool_result = await self.mcp_server.execute_tool(
                              tool_name,
                              tool_input
                          )

                          print(f"   ‚úď Tool returned: {json.dumps(tool_result, indent=2)[:200]}...")

                          tool_results.append({
                              "type": "tool_result",
                              "tool_use_id": content_block.id,
                              "content": json.dumps(tool_result)
                          })

                  # Add Claude's response and tool results to conversation
                  messages.append({
                      "role": "assistant",
                      "content": response.content
                  })
                  messages.append({
                      "role": "user",
                      "content": tool_results
                  })

                  iteration += 1
                  continue

              # Claude is done - no more tools needed
              elif response.stop_reason == "end_turn":
                  # Extract final answer from Claude
                  final_text = ""
                  for content_block in response.content:
                      if hasattr(content_block, "text"):
                          final_text += content_block.text

                  return {
                      "response": final_text,
                      "tool_calls_made": iteration
                  }

              iteration += 1

          return {
              "error": "Max iterations reached",
              "response": "Could not complete request"
          }

  # Backend/app/main.py

  @app.post("/api/smart-practice/next-question")
  async def get_next_question_mcp(
      request: Dict,
      current_user: StudentUser = Depends(get_current_student),
      db: Session = Depends(get_db)
  ):
      """
      Let Claude + MCP decide the next question.
      """

      # Initialize MCP server and client
      mcp_server = EducationMCPServer(db)
      mcp_client = MCPClient(mcp_server)

      # Build context for Claude
      prompt = f"""
      Find the optimal next practice question for student ID {current_user.id}.
      
      Context:
      - Session: {request.get('session_id')}
      - Just completed question: {request.get('last_question_id')}
      - Result: {'correct' if request.get('correct') else 'incorrect'}
      - Time spent: {request.get('time_spent')} seconds
      
      Your task:
      1. Understand the student's current knowledge state
      2. Determine what they should practice next
      3. Find an appropriate question at optimal difficulty
      4. Return the question ID and explain your reasoning
      
      Return your answer in this JSON format:
      {{
          "question_id": <id>,
          "reasoning": "<why this question>",
          "learning_objective": "<what skill this develops>"
      }}
      """

      # Claude decides what tools to use!
      result = await mcp_client.query(prompt)

      # Parse Claude's response
      try:
          response_json = json.loads(result["response"])
          question_id = response_json["question_id"]

          # Fetch the actual question
          question = db.query(NcertExercises).filter(
              NcertExercises.id == question_id
          ).first()

          return {
              "question": {
                  "id": question.id,
                  "text": question.question_text,
                  "difficulty": question.difficulty
              },
              "reasoning": response_json["reasoning"],
              "learning_objective": response_json["learning_objective"],
              "tool_calls_made": result["tool_calls_made"]
          }
      except:
          return {"error": "Could not parse Claude's response", "raw": result["response"]}

  What Actually Happens (Step-by-Step)

  1. Student requests next question ‚Üí Your FastAPI endpoint
  2. You send prompt to Claude via mcp_client.query() with tool definitions
  3. Claude reads the prompt and sees available tools
  4. Claude decides: "I need to understand this student first"
  5. Claude calls: get_student_profile(user_id=123)
  6. Your code executes the tool and returns data
  7. Claude receives the profile data
  8. Claude thinks: "They're weak in factoring, let me check performance"
  9. Claude calls: analyze_recent_performance(user_id=123, n_recent=5)
  10. Your code executes and returns recent performance
  11. Claude thinks: "They're struggling, let me find easier questions"
  12. Claude calls: search_questions(skill="factoring", difficulty_min=0.2, difficulty_max=0.4)
  13. Your code executes and returns matching questions
  14. Claude decides: "Question #47 is perfect"
  15. Claude returns final answer with reasoning
  16. Your code returns the question to the frontend

  The Key Insight

  You provide the tools. Claude provides the intelligence.

  You're not writing if/else logic for question selection. Claude dynamically reasons about:
  - Which tools to call
  - In what order
  - How to interpret the results
  - Whether to call more tools
  - When it has enough information

  That's why it's called Model Context Protocol - Claude is the model making contextual decisions about tool usage!
